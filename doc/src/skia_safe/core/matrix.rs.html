704    pub fn map_origin(&self) -> Point {
705        let mut x = self.translate_x();
706        let mut y = self.translate_y();
707        if self.has_perspective() {
708            let mut w = self[Member::Persp2];
709            if w != 0.0 {
710                w = 1.0 / w;
711            }
712            x *= w;
713            y *= w;
714        }
715        Point::new(x, y)
716    }
