82    #[test]
83    fn resource_provider_and_font_mgr_get_dropped_after_drop_of_dom() {
84        let svg = r##"<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 100 100" height = "256" width = "256">
85            </svg>"##;
86        let mut surface = surfaces::raster_n32_premul((256, 256)).unwrap();
87        let canvas = surface.canvas();
88        let font_mgr = FontMgr::new();
89        let provider: NativeResourceProvider = font_mgr.clone().into();
90        let dom = Dom::from_str(svg, provider.clone()).unwrap();
91        dom.render(canvas);
92        // Dom keeps the resource provider even after rendering.
93        assert!(provider.native()._ref_cnt() >= 2);
94        // And at least two of the font managers are referred to (one in the resource provider, and the other in the Dom)
95        assert!(font_mgr.native()._ref_cnt() >= 3);
96        drop(dom);
97        // now it's free.
98        assert_eq!(1, provider.native()._ref_cnt());
99        drop(provider);
100        // and so is the font mgr
101        assert_eq!(1, font_mgr.native()._ref_cnt());
102    }

179    #[cfg(feature = "save-svg-images")]
180    fn save_to_tmp(surface: &mut Surface, name: &str) {
181        use crate::EncodedImageFormat;
182        use std::{fs::File, io::Write, path::Path};
183
184        let image = surface.image_snapshot();
185        let data = image.encode(None, EncodedImageFormat::PNG, None).unwrap();
186        write_file(data.as_bytes(), Path::new(&format!("/tmp/svg-{name}.png")));
187
188        pub fn write_file(bytes: &[u8], path: &Path) {
189            let mut file = File::create(path).expect("failed to create file");
190            file.write_all(bytes).expect("failed to write to file");
191        }
192    }
